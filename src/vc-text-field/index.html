Material 2 Text Field Web Component Demo

Basic Usage

<vc-text-field 
  label="Username" 
  helper-text="Enter your username">
</vc-text-field>

Required Field

<vc-text-field 
  label="Email" 
  required 
  helper-text="We'll never share your email">
</vc-text-field>

Password Field

<vc-text-field 
  label="Password" 
  type="password" 
  helper-text="At least 8 characters">
</vc-text-field>

With Character Counter

<vc-text-field 
  label="Bio" 
  max-length="50" 
  helper-text="Tell us about yourself">
</vc-text-field>

Disabled Field

<vc-text-field 
  label="Username" 
  value="johndoe" 
  disabled>
</vc-text-field>

Read-only Field

<vc-text-field 
  label="User ID" 
  value="USR12345" 
  readonly>
</vc-text-field>

With Error

<vc-text-field 
  label="Email" 
  value="invalid-email" 
  error-message="Please enter a valid email address">
</vc-text-field>

JavaScript API

// Get reference to the element
const textField = document.getElementById('js-field');

// Set value programmatically
textField.value = 'John Doe';

// Get current value
console.log(textField.value);

// Set focus
textField.focus();

// Check validity
console.log(textField.checkValidity());
