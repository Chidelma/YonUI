JavaScript Usage

// Create a FAB element programmatically
const fab = document.createElement('vc-fab');
fab.icon = 'add';
fab.color = 'primary';
fab.extended = true;
fab.label = 'Create';
document.body.appendChild(fab);

// Listen for click events
fab.addEventListener('fab-click', (e) => {
  console.log('FAB clicked!', e.detail);
});

// Disable the FAB
fab.disabled = true;

// Enable the FAB
fab.disabled = false;
