Usage Example

<!-- Include the Web Component -->
<script src="vc-btn.js"></script>

<!-- Use the button -->
<vc-btn variant="contained" color="primary">Click Me</vc-btn>

<!-- Outlined variant -->
<vc-btn variant="outlined" color="secondary">Outlined</vc-btn>

<!-- Text variant -->
<vc-btn variant="text" color="default">Text Button</vc-btn>

<!-- Disabled state -->
<vc-btn variant="contained" color="primary" disabled>Disabled</vc-btn>

<!-- Different sizes -->
<vc-btn variant="contained" color="primary" size="small">Small</vc-btn>
<vc-btn variant="contained" color="primary" size="medium">Medium</vc-btn>
<vc-btn variant="contained" color="primary" size="large">Large</vc-btn>
