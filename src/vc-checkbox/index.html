Material 2 Checkbox Web Component

Basic Usage

<vc-checkbox label="Unchecked checkbox"></vc-checkbox> <vc-checkbox checked label="Checked checkbox"></vc-checkbox> <vc-checkbox indeterminate label="Indeterminate checkbox"></vc-checkbox> <vc-checkbox disabled label="Disabled checkbox"></vc-checkbox> <vc-checkbox checked disabled label="Disabled checked checkbox"></vc-checkbox>

Interactive Example

// JavaScript const checkbox = document.getElementById('interactive-checkbox'); // Listen for change events checkbox.addEventListener('change', (event) => { console.log('Checkbox changed:', event.detail.checked); }); // Using the API methods document.getElementById('toggle-checked').addEventListener('click', () => { checkbox.checked = !checkbox.checked; }); document.getElementById('toggle-indeterminate').addEventListener('click', () => { checkbox.indeterminate = !checkbox.indeterminate; }); document.getElementById('toggle-disabled').addEventListener('click', () => { checkbox.disabled = !checkbox.disabled; }); document.getElementById('change-label').addEventListener('click', () => { checkbox.label = 'Label changed at ' + new Date().toLocaleTimeString(); });

Form Integration

<form id="demo-form"> <vc-checkbox id="terms-checkbox" label="I agree to terms and conditions" value="agreed"></vc-checkbox> <vc-checkbox id="newsletter-checkbox" label="Subscribe to newsletter" value="subscribe"></vc-checkbox> <button type="submit">Submit</button> </form> // JavaScript document.getElementById('demo-form').addEventListener('submit', (event) => { event.preventDefault(); const termsChecked = document.getElementById('terms-checkbox').checked; const newsletterChecked = document.getElementById('newsletter-checkbox').checked; console.log('Form submitted:', { termsAccepted: termsChecked, newsletterSubscribed: newsletterChecked }); // You can also get the values if needed if (termsChecked) { console.log('Terms value:', document.getElementById('terms-checkbox').value); } });
